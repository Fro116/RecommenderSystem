CREATE TABLE collect_users (
    source TEXT,
    username TEXT,
    userid BIGINT,
    fingerprint TEXT,
    data BYTEA,
    db_refreshed_at DOUBLE PRECISION
);
CREATE INDEX collect_users_source_username_idx ON collect_users (source, lower(username));

CREATE TABLE inference_users (
    source TEXT,
    username TEXT,
    userid BIGINT,
    fingerprint TEXT,
    data BYTEA,
    db_refreshed_at DOUBLE PRECISION
);
CREATE INDEX inference_users_source_username_idx ON inference_users (source, lower(username));

CREATE TABLE autocomplete_users (
    source TEXT,
    prefix TEXT,
    data BYTEA
);
CREATE INDEX autocomplete_users_source_prefix_idx ON autocomplete_users (source, prefix);
